222  /**
223  * @brief Class combining multiple iterators into one. Use it to iterate over multiple ranges at the same time.
224  * @details @copybrief
225  * ZipIterators only support the operators of the least powerful underling iterator. Zipping a random access
226  * iterator (e.g. from std::vector) and a bidirectional iterator (e.g. from std::list) results in a
227  * bidirectional iterator. ZipIterators return a tuple of references to the range elements. When using
228  * structured bindings, no additional reference binding is necessary.
229  *
230  * Let ```z``` be a ZipIterator composed from two ```std::vector<int>```
231  * ```
232  * auto [val1, val2] = *z; // val1 and val2 are references to the vector elements
233  * val1 = 17; // this will change the respective value in the first vector
234  * ```
235  * @tparam Iterators Underlying iterator types
236  */

242  using pointer = void;
243  using difference_type = std::ptrdiff_t;

247  : iterators(iterators) {}

533  struct ZipView {

539  template<bool Const>
540  using Sentinels = std::tuple<decltype(std::end(

542  using IteratorTuple = Iterators<false>;
543  using CIteratorTuple = Iterators<true>;
