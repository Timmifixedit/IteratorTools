105 /**
106 * @brief Proxy reference type that supports assignment and swap even on const instances.
107 * @details @copybrief
108 * Actual constness is determined by element constness
109 * @tparam Ts
110 */
111 template<typename ...Ts>
112 struct RefTuple : public std::tuple<Ts...> {
113 using std::tuple<Ts...>::tuple;
114
115 static constexpr bool Assignable = (not std::is_const_v<std::remove_reference_t<Ts>> && ...);
116
117 template<typename Dummy = int>
118 constexpr auto operator=(RefTuple &&other) noexcept((std::is_nothrow_move_assignable_v<std::remove_reference_t<Ts>> && ...))
119 -> std::enable_if_t<std::is_same_v<Dummy, int> and Assignable, RefTuple &> {
120 if (selfAssignGuard(other)) {
121 return *this;
122 }
123
124 moveAssign(*this, other);
125 return *this;
126 }
127
128 template<typename ...Us>
129 decltype(auto) operator=(const std::tuple<Us...> &other) const {
130 if (selfAssignGuard(other)) {
131 return *this;
132 }
133
134 copyAssign(*this, other);
135 return *this;
136 }
137
138 template<typename ...Us>
139 decltype(auto) operator=(std::tuple<Us...> &&other) const {
140 if (selfAssignGuard(other)) {
141 return *this;
142 }
143
144 moveAssign(*this, other);
145 return *this;
146 }

168 template<typename Tuple>
169 constexpr auto swap(Tuple &&other) const -> std::enable_if_t<std::is_same_v<Tuple, RefTuple> and Assignable> {
170 auto tmp = std::move(*this);
171 // move of forwarding reference because we always move, even const ref.
172 moveAssign(*this, std::move(other));

428 /**
429 * @brief CRTP-class that provides additional pointer arithmetic operators synthesized from basic operators
430 * @details @copybrief
431 * Adds the following operators
432 * - postfix increment and decrement (requires the respective prefix operators)
433 * - array subscript operator[] (requires operator+ and dereference operator)
434 * - binary arithmetic operators (requires compound assignment operators)
435 * - inequality comparison (requires operator==)
436 * - less than or equal comparison (requires operator>)
437 * - greater than or equal comparison (requires operator<)
438 * @tparam Impl Base class
439 */
440 template<typename Impl>
441 struct SynthesizedOperators {
442
443 /**
444 * Array subscript operator
445 * @tparam Implementation SFINAE helper, do not specify explicitly
446 * @param n index
447 * @return *(*this + n)
448 */
449 template<REQUIRES_IMPL(Impl, *(INSTANCE_OF_IMPL + INSTANCE_OF(typename Implementation::difference_type)))>
450 constexpr decltype(auto) operator[](typename Implementation::difference_type n) const
451 noexcept(noexcept(*(std::declval<Impl>() + n))) {
452 return *(this->getImpl() + n);
453 }
454
455 /**
456 * @copydoc SynthesizedOperators::operator[](typename Implementation::difference_type n)
457 */
458 template<REQUIRES_IMPL(Impl, *(INSTANCE_OF_IMPL + INSTANCE_OF(typename Implementation::difference_type)))>
459 constexpr decltype(auto) operator[](typename Implementation::difference_type n)
460 noexcept(noexcept(*(std::declval<Impl>() + n))) {
461 return *(this->getImpl() + n);
462 }
463
464 /**
465 * Postfix increment. Synthesized from prefix increment
466 * @tparam Implementation SFINAE helper, do not specify explicitly
467 * @return Instance of Impl
468 */
469 template<REQUIRES_IMPL(Impl, ++INSTANCE_OF_IMPL)>
470 constexpr Impl operator++(int)
471 noexcept(noexcept(++std::declval<Impl>()) && std::is_nothrow_copy_constructible_v<Impl>) {
472 auto tmp = this->getImpl();
473 this->getImpl().operator++();
474 return tmp;
475 }
476
477 /**
478 * Postfix decrement. Synthesized from prefix decrement
479 * @tparam Implementation SFINAE helper, do not specify explicitly
480 * @return Instance of Impl
481 */
482 template<REQUIRES_IMPL(Impl, --INSTANCE_OF_IMPL)>
483 constexpr Impl operator--(int)
484 noexcept(noexcept(--std::declval<Impl>()) && std::is_nothrow_copy_constructible_v<Impl>) {
485 auto tmp = this->getImpl();
486 this->getImpl().operator--();
487 return tmp;
488 }
489
490 /**
491 * Binary +plus operator. Synthesized from compound assignment operator+=
492 * @tparam Implementation SFINAE helper, do not specify explicitly
493 * @param it left hand side
494 * @param n right hand side
495 * @return Instance of Impl
496 */
497 template<REQUIRES_IMPL(Impl, INSTANCE_OF_IMPL += INSTANCE_OF(typename Implementation::difference_type)) >
498 friend constexpr auto operator+(Impl it, typename Implementation::difference_type n)
499 noexcept(noexcept(std::declval<Impl>() += n)) {
500 it += n;
501 return it;
502 }
503
504 /**
505 * Binary +plus operator. Synthesized from compound assignment operator+=
506 * @tparam Implementation SFINAE helper, do not specify explicitly
507 * @param n left hand side
508 * @param it right hand side
509 * @return Instance of Impl
510 */
511 template<REQUIRES_IMPL(Impl, INSTANCE_OF_IMPL += INSTANCE_OF(typename Implementation::difference_type))>
512 friend constexpr auto operator+(typename Implementation::difference_type n, Impl it)
513 noexcept(noexcept(std::declval<Impl>() += n)) {
514 it += n;
515 return it;
516 }
517
518 /**
519 * Binary minus operator. Synthesized from compound assignment operator-=
520 * @tparam Implementation SFINAE helper, do not specify explicitly
521 * @param it left hand side
522 * @param n right hand side
523 * @return Instance of Impl
524 */
525 template<REQUIRES_IMPL(Impl, INSTANCE_OF_IMPL -= INSTANCE_OF(typename Implementation::difference_type)) >
526 friend constexpr auto operator-(Impl it, typename Implementation::difference_type n)

535 * @tparam Implementation SFINAE helper, do not specify explicitly

581 /**
582 * @brief Class combining multiple iterators into one. Use it to iterate over multiple ranges at the same time.
583 * @details @copybrief
584 * ZipIterators only support the operators of the least powerful underling iterator. Zipping a random access
585 * iterator (e.g. from std::vector) and a bidirectional iterator (e.g. from std::list) results in a
586 * bidirectional iterator. All operators are SFINAE friendly.
587 *
588 * ZipIterators return a tuple of references to the range elements. When using
589 * structured bindings, no additional reference binding is necessary.
590 *
591 * Let ```z``` be a ZipIterator composed from two ```std::vector<int>```
592 * ```
593 * auto [val1, val2] = *z; // val1 and val2 are references to the vector elements
594 * val1 = 17; // this will change the respective value in the first vector
595 * ```
596 * @tparam Iterators Underlying iterator types
597 */
598 template<typename Iterators>
599 class ZipIterator
600 : public traits::iterator_category_from_value<traits::minimum_category_v<Iterators>>,
601 public SynthesizedOperators<ZipIterator<Iterators>> {
602
603 public:
604 using value_type = traits::values_t<Iterators>;
605 using reference = traits::references_t<Iterators>;
606 using pointer = void;
607 using difference_type = std::ptrdiff_t;
608
609 private:
610 BINARY_TUPLE_FOR_EACH_FOLD(ELEMENT1 == ELEMENT2, ||, oneEqual)
611 BINARY_TUPLE_FOR_EACH_FOLD(ELEMENT1 < ELEMENT2, &&, allLess)
612 BINARY_TUPLE_FOR_EACH_FOLD(ELEMENT1 > ELEMENT2, &&, allGreater)
613 BINARY_TUPLE_FOR_EACH(std::min<difference_type>({ELEMENT1 - ELEMENT2 ...}), minDifference)
614 Iterators iterators;
615
616 public:
617 using SynthesizedOperators<ZipIterator>::operator++;
618 using SynthesizedOperators<ZipIterator>::operator--;
619
620 constexpr ZipIterator() noexcept = default;
621
622 explicit constexpr ZipIterator(
623 const Iterators &iterators) noexcept(std::is_nothrow_copy_constructible_v<Iterators>)
624 : iterators(iterators) {}

626 /**
627 * Increments all underlying iterators by one
628 * @tparam Its SFINAE guard, do not specify
629 * @return reference to this
630 */
631 template<typename Its = Iterators, typename = std::enable_if_t<traits::is_incrementable_v<Its>>>
632 constexpr ZipIterator &operator++() noexcept(traits::is_nothrow_incrementible_v<Iterators>) {
633 std::apply([](auto &&...it) { (++it, ...); }, iterators);
634 return *this;
635 }

640 * access
641 */
642 ///@{

644 /**
645 * Decrements all underlying iterators by one. Only available if all iterators support at least
646 * bidirectional access
647 * @tparam IsBidirectional SFINAE guard, do not specify
648 * @return reference to this

909 struct CounterIterator : public SynthesizedOperators<CounterIterator<T>> {
910 using value_type = T;
911 using reference = T;
912 using pointer = void;
913 using iterator_category = std::random_access_iterator_tag;

915 static_assert(std::is_integral_v<T> && !std::is_floating_point_v<T>);

927 counter(begin), increment(increment) {}

969 /**
970 * Difference between two CounterIterators
971 * @param other right hand side
972 * @return integer ```n``` with the smallest possible absolute value such that ```other + n <= *this```
973 * @note When other has the same increment as ```*this```, then the returned value is guaranteed to
974 * fulfil ```other + n == *this```. In the following example, this is not the case:
975 * ```
976 * CounterIterator a(8, 1);
977 * CounterIterator b(4, 3);
978 * auto diff = a - b; // yields 1 since b + 1 <= a
979 * ```
980 */
981 constexpr difference_type operator-(const CounterIterator &other) const noexcept {
982 return static_cast<difference_type>((counter - other.counter) / other.increment);

1016 * @return true if
